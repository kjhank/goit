    Właściwość transition-timing-function

    transition-timing-function: ease | ease-in | ease-out | ease-in-out | linear | step-start | step-end | steps() | cubic-bezier()

    Wartość linear

    div {
  transition-duration: 3s;
  transition-property: translate;
  transition-timing-function: linear
}

    Wartość ease-in

    div {
  transition-duration: 3s;
  transition-property: translate;
  transition-timing-function: ease-in
}

    Wartość ease-out

    div {
  transition-duration: 3s;
  transition-property: translate;
  transition-timing-function: ease-out
}

    Wartość ease-in-out

    div {
  transition-duration: 3s;
  transition-property: translate;
  transition-timing-function: ease-in-out
}

    Wartość steps()

    div {
  transition-duration: 3s;
  transition-property: translate;
  transition-timing-function: steps(10, jump-start);
}

    Wartość cubic-bezier()

    div {
  transition-duration: 3s;
  transition-property: translate;
  transition-timing-function: cubic-bezier(0.83, 0, 0.17, 1);
}
