    $gridWidth: 600px;
$gridGutter: 30px;

.row {
  display: inline-block;
  width: $gridWidth;
  white-space: nowrap;
}

.col {
  display: inline-block;
  white-space: normal;
}

.col-1 {
  width: $gridWidth;
}

.col-2 {
  width: ($gridWidth - 2 * $gridGutter) / 2;
  margin: $gridGutter / 2;
}

.col-3 {
  width: ($gridWidth - 2 * $gridGutter) / 3;
  margin: $gridGutter / 3;
}
