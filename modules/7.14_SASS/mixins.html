    Reużywalne funkcje (@mixin)

    @mixin name {
// declarations
}

selector {
@include name;
}

    Użycie @mixin jak @extend

    @mixin reset-list {
  margin: 0;
  padding: 0;
  list-style: none;
}

nav ul {
  @include reset-list;
}
            
    nav ul {
  margin: 0;
  padding: 0;
  list-style: none;
}

    Parametry @mixin

    @mixin border($color, $width) {
  border-width: $width;
  border-color: $color;
}

.bordered {
  border-style: solid;

  &--rebeccapurple {
    @include border(rebeccapurple, 0.2em);
  }

  &--limegreen {
    @include border(limegreen, 0.3em);
  }
}
            
    .bordered {
  border-style: solid;
}
.bordered--rebeccapurple {
  border-width: 0.2em;
  border-color: rebeccapurple;
}
.bordered--limegreen {
  border-width: 0.3em;
  border-color: limegreen;
}

    Domyślne parametry @mixin

    @mixin flex($direction: row, $justify: flex-start, $align: center) {
  display: flex;
  flex-direction: $direction;
  justify-content: $justify;
  align-items: $align;
}

.flex {
  @include flex;
}

.flex-end {
  @include flex(row-reverse, flex-end);
  gap: 2em;
}
            
    .flex {
  display: flex;
  flex-direction: row;
  justify-content: flex-start;
  align-items: center;
}

.flex-end {
  display: flex;
  flex-direction: row-reverse;
  justify-content: flex-end;
  align-items: center;
  gap: 2em;
}
            
    @mixin flex($direction: row, $justify: flex-start, $align: center) {
  display: flex;
  flex-direction: $direction;
  justify-content: $justify;
  align-items: $align;
}

.flex {
  @include flex;
}

.flex-align-end {
  @include flex($align: flex-end);
}
            
    .flex {
  display: flex;
  flex-direction: row;
  justify-content: flex-start;
  align-items: center;
}

.flex-align-end {
  display: flex;
  flex-direction: row;

  align-items: flex-end;
