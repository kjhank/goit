    Użycie interpolacji wartości w pętli @each

    $sizes: 40px, 50px, 80px;

@each $size in $sizes {
  .icon-#{$size} {
    font-size: $size;
    height: $size;
    width: $size;
  }
}
            
    .icon-40px {
  font-size: 40px;
  height: 40px;
  width: 40px;
}

.icon-50px {
  font-size: 50px;
  height: 50px;
  width: 50px;
}

.icon-80px {
  font-size: 80px;
  height: 80px;
  width: 80px;
}
            
    $animals: dog, cat, bird, turtle;

@each $animal in $animals {
  .#{$animal}-icon {
    background-image: url('../images/#{$animal}.png');
  }
}
            
    .dog-icon {
  background-image: url("../images/dog.png");
}

.cat-icon {
  background-image: url("../images/cat.png");
}

.bird-icon {
  background-image: url("../images/bird.png");
}

.turtle-icon {

}

    Mapy (słowniki)

    @$font-weights: (
	"regular": 400,
	"medium": 500,
	"bold": 700
);

.font {
	&--regular {
		font-weight: map-get($font-weights, regular);
	}

	&--medium {
		font-weight: map-get($font-weights, medium);
	}

	&--bold {
		font-weight: map-get($font-weights, bold);
	}
}
            
    .font--regular {
  font-weight: 400;
}
.font--medium {
  font-weight: 500;
}
.font--bold {
  font-weight: 700;
}

    Połączenie pętli @each i słowinków

    $font-weights: (
  "regular": 400,
  "medium": 500,
  "bold": 700
);

.font {
  @each $name, $weight in $font-weights {
    &--#{$name} {
      font-weight: $weight;
    };
  }
}
            
    .font--regular {
  font-weight: 400;
}
.font--medium {
  font-weight: 500;
}
.font--bold {
  font-weight: 700;
}
