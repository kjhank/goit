    Pseudoklasy :enabled oraz :disabled

    <form action="#" class="example-form" name="form-required">
  <label for="myName">Imię</label>
  <input id="myName" name="userName" required type="text" />
  <label for="myPass">Hasło</label>
  <input disabled id="myPass" name="password" required type="password" />
  <button type="submit">Wyślij</button>
  <p class="visually-hidden form-message" data-message>komunikat</p>
</form>
          
    form input:disabled {
  rotate: z -0.01turn;
  cursor: not-allowed;
}

form input:enabled {
  filter: drop-shadow(0 0 1em var(--color-sub));
}

form input:disabled:hover {
  filter: drop-shadow(0 0 1em var(--color-main));
}

    Pseudoklasa :invalid

    <form action="#" class="example-form example-form--invalid" name="form-invalid">
  <label for="username">Nazwa użytkownika (2-127 łacińskich liter)</label>
  <input type="text" name="username" id="username" pattern="([a-zA-Z]){2,127}" placeholder="andRzej" required />
  <label for="myPassword">Hasło (5-255 znaków alfanumerycznych lub <code>!@#$%^&*()_+-=</code>)</label>
  <div class="password-wrapper">
    <input type="password" name="myPassword" id="myPassword" pattern="[\w!@#$%^&*()_+-=]{5,255}"
      placeholder="D#@asd666" required />
    <button type="button">poka</button>
  </div>
  <button type="submit">Wyślij</button>
  <p class="visually-hidden form-message" data-message>komunikat</p>
</form>
            
    input:not(:placeholder-shown):not(:focus):invalid {
  background-color: red;
}

    Przykładowe użycie pseudoklas :invalid wraz z :placeholder-shown i :focus

    Pseudoklasy :out-of-range :in-range

    <form action="#" class="example-form example-form--range" name="form-range">
  <label for="deliveryDate">Data dostawy</label>
  <input required id="deliveryDate" min="2023-06-08" name="deliveryDate" type="date" value="2023-06-07" />
  <label for="count">Liczba sztuk</label>
  <input required min="1" id="count" name="count" type="number" value="0" />
  <button type="submit">Wyślij</button>
</form>
            
    input:out-of-range {
  background-color: red;
}

input:in-range {
  background-color: green;
}

    Przykładowe użycie pseudoklas :in-range oraz :out-of-range
